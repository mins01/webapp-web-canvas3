툴 라이프 사이클

ready() // 툴을 사용할 상태로 준비시킴 (라이프 사이클 아님)

construct()
  init() // 최초 한번만 불린다.

activate() // with ready(), 툴이 선택되면 호출
  

pointerdown() // draw()
pointermove() // draw()
pointerup() // draw()

confirm() // with ready(), 동작을 적용시킴
cancel() // with ready(), 동작이 취소되면
deactivate() // 툴이 다른 툴로 바뀔 경우

draw() // 페인팅 동작
---------=-----------
다시

# 툴 선택시 / 레이어 등의 변경시
activate() -> ready()

# 마우스 이동 (start,update,end 는 후킹용)
pointerdown() -> start() -> ?draw()
pointermove() -> update() -> ?draw()
pointerup() -> end() -> ?draw()

# 적용과 취소가 있을 경우
confirm()/cancel()

# 툴이 선택이 취소될 경우
deactivate() // 내용 복구.
